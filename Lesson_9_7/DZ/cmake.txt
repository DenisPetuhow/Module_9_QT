Вариант 3: Использовать статическую линковку (ПРОЩЕ!)
Если .dll нет — не мучайся, используй статическое подключение:
cmakecmake_minimum_required(VERSION 3.5)
project(PreGraphic VERSION 0.1 LANGUAGES CXX)

set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Флаг для MinGW (решает ошибку "too many sections")
if(MINGW)
    add_compile_options(-Wa,-mbig-obj)
endif()

find_package(Qt6 REQUIRED COMPONENTS Widgets Concurrent PrintSupport)

set(PROJECT_SOURCES
        main.cpp
        mainwindow.cpp
        mainwindow.h
        mainwindow.ui
        graphwindow.cpp
        graphwindow.h
        # ===== СТАТИЧЕСКОЕ ПОДКЛЮЧЕНИЕ =====
        qcustomplot.cpp
        qcustomplot.h
)

add_executable(${PROJECT_NAME}
    ${PROJECT_SOURCES}
)

# БЕЗ add_compile_definitions(QCUSTOMPLOT_USE_LIBRARY) !
# БЕЗ qcustomplot2.dll !

target_link_libraries(${PROJECT_NAME}
    Qt6::Widgets
    Qt6::Concurrent
    Qt6::PrintSupport
)
